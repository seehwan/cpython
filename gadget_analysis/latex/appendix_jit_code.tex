def optimized_jit_func_0(x):
    # High-gadget-density function:
    #   - Heavy COMPARE_OP usage (if/elif chains)
    #   - Nested helper calls (CALL stencils)
    #   - Dict ops (STORE_SUBSCR_DICT / BINARY_SUBSCR)
    #   - Object attribute access (LOAD/STORE_ATTR)
    #   - Arithmetic mix (BINARY_OP variants)

    def helper(a, b, c):
        if a > b:
            result = a ^ c
        elif a < b:
            result = b ^ c
        else:
            result = a + b + c
        if result > 0x000000C3:
            result -= 1
        if result < 0:
            result += 1
        return result & 0xFFFFFFFF

    data = {}
    class Container:
        val1 = 0
        val2 = 0
        val3 = 0
    obj = Container()
    acc = x

    for i in range(5000):
        if i % 10 == 0:
            acc = helper(acc, i, 0x000000C3)
        if i % 20 == 0:
            acc = helper(acc ^ 0x000000C3, i & 0xFF, acc >> 3)

        acc = (acc + 0x000000C3) & 0xFFFFFFFF
        acc = (acc * 3) & 0xFFFFFFFF
        acc ^= i << (i % 8)
        acc = ((acc << 5) | (acc >> 27)) & 0xFFFFFFFF

        if i % 50 == 0:
            data[i] = acc & 0xFF
            acc ^= data.get(i - 50, 0)

        if i % 100 == 0:
            obj.val1 = acc & 0xFFFF
            obj.val2 = (acc >> 16) & 0xFFFF
            acc += obj.val1 ^ obj.val2

        if acc > 0x7FFFFFFF:
            acc &= 0x7FFFFFFF
        if acc < 0x000000C3:
            acc += i
        elif acc > (0x000000C3 << 10):
            acc -= i

        mod = i % 7
        if mod == 0:
            acc ^= 0x12345678
        elif mod == 1:
            acc += 0x87654321
        elif mod == 2:
            acc *= 5
        elif mod == 3:
            acc >>= 3
        elif mod == 4:
            acc <<= 2
        elif mod == 5:
            acc |= 0x000000C3
        else:
            acc &= 0xFFFFFFFF

        if acc > 0x000000C3 * 2:
            if acc > 0x000000C3 * 4:
                acc -= 0x000000C3
            else:
                acc += i

    return acc
